add_namespace = wur

#### Darkest Day Trigger ####
country_event = {
	id = wur.1

	hidden = yes

	fire_only_once = yes

	trigger = {
		country_exists = GER
		date > 1932.5.1
	}
	
	immediate = {
		GER = {
			news_event = {
				id = wur.2
			}
		}
	}

	option = {
		name = wur.1.a
	}

}

#### Darkest Day Event ####
news_event = {
	id = wur.2
	title = wur.2.t
	desc = wur.2.d
	picture = GFX_darkestday1
	is_triggered_only = yes

	option = {
		name = wur.2.a
		add_ideas = WUR_darkest_day
		add_political_power = -20
		add_popularity = {
			ideology = revolutionary_socialist popularity = 0.05
		}
		hidden_effect = {
			country_event = {
				id = wur.3 days = 1
			}
		}
	}
}
