### very_high_popularity - Very High Popularity
country_event = {
	id = sp.political.11
	title = sp.political.11.t
	desc = sp.political.11.d
	picture = popularity_rising

	is_triggered_only = yes

	trigger = {
		NOT = { 
			has_idea = very_high_popularity
		}
		check_variable = { NS_ruling_party_pop > 79.99 }
	}

	option = {
		name = sp.political.11.a
		if = {
			limit = { has_idea = very_low_popularity }
			effect_tooltip = { swap_ideas = { remove_idea = very_low_popularity add_idea = very_high_popularity } }
			hidden_effect = {  swap_ideas = { remove_idea = very_low_popularity add_idea = low_popularity } }
		}
		if = {
			limit = { has_idea = low_popularity }
			effect_tooltip = { swap_ideas = { remove_idea = low_popularity add_idea = very_high_popularity } }
			hidden_effect = {  swap_ideas = { remove_idea = low_popularity add_idea = normal_popularity } }
		}
		if = {
			limit = { has_idea = normal_popularity }
			effect_tooltip = { swap_ideas = { remove_idea = normal_popularity add_idea = very_high_popularity } }
			hidden_effect = {  swap_ideas = { remove_idea = normal_popularity add_idea = high_popularity } }
		}
		if = {
			limit = { has_idea = high_popularity }
			swap_ideas = { remove_idea = high_popularity add_idea = very_high_popularity }
		}
	}

}
