country_event = { #Very Good Stability -> Good Stability
	id = sp.political.1
	title = sp.political.1.t
	desc = sp.political.1.d
	picture = GFX_report_event_generic_rally_3

	trigger = {
		has_stability < 0.8
		has_idea = very_good_stability
	}

	mean_time_to_happen = {
		months = 1

		modifier = {
			factor = 1
			has_stability < 0.8
			has_stability > 0.6
		}
	}

	immediate = {
		hidden_effect = {
			# Need a double swap here to keep the ordering in the display
			swap_ideas = {
				remove_idea = very_good_stability
				add_idea = good_stability
			}
		}
	}

	option = {
		name = sp.political.1.a

		effect_tooltip = {
			swap_ideas = {
				remove_idea = very_good_stability
				add_idea = good_stability
			}
		}
	}
}

			}

				revolutionary_socialist > 0.799
			}
			AND = {
				has_government = social_democrat
				social_democrat > 0.799
			}
			AND = {
				has_government = social_liberal
				social_liberal > 0.799
			}
			AND = {
				has_government = liberal_conservative
				liberal_conservative > 0.799
			}
			AND = {
				has_government = social_conservative
				social_conservative > 0.799
			}

				has_government = authoritarian_democrat
				authoritarian_democrat > 0.799
			}
			AND = {
				has_government = right_wing_populist
				right_wing_populist > 0.799
			}
			AND = {
				has_government = national_syndicalist
				national_syndicalist > 0.799
			}
		}
	}

	mean_time_to_happen = {
		days = 2
	}

	option = {
		name = sp.political.11.a
		if = {
			limit = { has_idea = very_low_popularity }
			effect_tooltip = { swap_ideas = { remove_idea = very_low_popularity add_idea = very_high_popularity } }
			hidden_effect = {  swap_ideas = { remove_idea = very_low_popularity add_idea = low_popularity } }

			limit = { has_idea = low_popularity }
			effect_tooltip = { swap_ideas = { remove_idea = low_popularity add_idea = very_high_popularity } }
			hidden_effect = {  swap_ideas = { remove_idea = low_popularity add_idea = normal_popularity } }
		}
		if = {
			limit = { has_idea = normal_popularity }
			effect_tooltip = { swap_ideas = { remove_idea = normal_popularity add_idea = very_high_popularity } }
			hidden_effect = {  swap_ideas = { remove_idea = normal_popularity add_idea = high_popularity } }
		}
		if = {
			limit = { has_idea = high_popularity }
			swap_ideas = { remove_idea = high_popularity add_idea = very_high_popularity }
		}
	}

}

	id = sp.political.16
	title = sp.political.16.t
	desc = sp.political.16.d

	trigger = {
		NOT = {
			has_government = leninist
			has_government = marxist
			has_government = revolutionary_socialist
			has_government = democratic_socialist
		}
		OR = {
			AND = { has_idea_with_trait = leninist leninist < 0.7 }
			AND = { has_idea_with_trait = marxist marxist < 0.7 }
			AND = { has_idea_with_trait = revolutionary_socialist revolutionary_socialist < 0.7 }
			AND = { has_idea_with_trait = democratic_socialist democratic_socialist < 0.7 }
		}

	mean_time_to_happen = {
		months = 18

			factor = 2

			factor = 4
			has_idea_with_trait = leninist
			leninist > 0.399

			factor = 2
